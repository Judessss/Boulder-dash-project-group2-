1   package entity.element.aliveElement;

3   import entity.Map;
4   import entity.element.Direction;
5   import entity.element.Position;
6   import entity.element.Sprite;
7   import entity.element.TraversableByAlive;
8   import entity.element.TraversableByFalling;
9   
10  /**
11   * The Mob class
12   *
13   * @author BIRAHEKA
14   * @version 1.0
15   */
16  public class Mob extends AliveElement {
17  
18  	/**
19  	 * The path to the mob.
20  	 */
21  	private static final String spritePath = "/sprites/mobs/";
22  	private static final String imageName = "mob.gif";
23  	private static final char sprite_ref = 'M';
24  	private static final TraversableByAlive traversableByAlive = TraversableByAlive.Mob;
25  	private static final TraversableByFalling traversableByFalling = TraversableByFalling.Alive;
26  	
27  	private static final Sprite sprite = new Sprite(sprite_ref, spritePath, imageName);
28  
29  	/**
30  	 * Instantiates a new mob.
31  	 * @param map The map
32  	 * @param position The position in the map
33  	 */
34  	
35  	public Mob(Map map, Position position) {
36  		super(map, position);
37  		sprite.loadImage();
38  		this.setSprite(sprite);	
39  		this.setTraversableByAlive(traversableByAlive);
40  		this.setTraversableByFalling(traversableByFalling);
41  	}
42  	
43  	/**
44  	 * Instantiates a new mob.
45  	 */
46  	public Mob() {
47  	}
48  
49  	/**
50  	 * Gives the direction on the left of the direction given.
51  	 * @param lastMove
52  	 * 				Last move of the mob {@link #getLastMove()}
53  	 * @return The next direction of the mob
54  	 */
55  	public Direction checkOnLeft(Direction lastMove) {
56  		switch(lastMove) {
57  		case Up:
58  			return Direction.Left;
59  		case Left:
60  			return Direction.Down;
61  		case Down:
62  			return Direction.Right;
63  		case Right:
64  			return Direction.Up;
65  		default:
66  			break;
67  		}
68  		return lastMove;
69  	}
70  
71  	/**
72  	 * Moves the mob :
73  	 * <ul>
74  	 *     <li>Randomly</li>
75  	 *     <li>By following the player</li>
76  	 * </ul>
77  	 */
78  	public void moveMobs() {
79  		Direction direction;
80  		if(this.checkAlivePermeability(this.getLastMove())==TraversableByAlive.Traversable || this.checkAlivePermeability(this.getLastMove())==TraversableByAlive.Player) {
81  			direction = this.getLastMove();
82  		} else if(this.checkAlivePermeability(this.checkOnLeft(this.getLastMove()))==TraversableByAlive.Traversable || this.checkAlivePermeability(this.checkOnLeft(this.getLastMove()))==TraversableByAlive.Player) {
83  			direction = this.checkOnLeft(this.getLastMove());
84  		} else if(this.checkAlivePermeability(this.checkOnLeft(this.checkOnLeft(this.getLastMove())))==TraversableByAlive.Traversable || this.checkAlivePermeability(this.checkOnLeft(this.checkOnLeft(this.getLastMove())))==TraversableByAlive.Player) {
85  			direction = this.checkOnLeft(this.checkOnLeft(this.getLastMove()));
86  		} else {
87  			direction = this.checkOnLeft(this.checkOnLeft(this.checkOnLeft(this.getLastMove())));
88  		}
89  		this.move(direction);
90  	}
91  
92  	/**
93  	 * Check the environment of the mob to know if he can move, then moves the mob,
94  	 * if it was a player on the way, kills the player.
95  	 * @param direction The {@link Direction}
96  	 */
97  	@Override
98  	public void move(Direction direction) {
99  		switch(this.checkAlivePermeability(direction)) {
100 		case Traversable:
101 			this.setLastMove(direction);
102 			super.move(direction);
103 			break;
104 		case Player:
105 			this.getMap().getPlayer().die();
106 			break;
107 		default:
108 			break;
109 		}
110 	}
111 
112 	/**
113 	 * Kills the mob.
114 	 */
115 	@Override
116 	public void die() {
117 		super.die();
118 		this.blow();
119 		this.getMap().getMobs().remove(this);
120 	}
121 
122 	/**
123 	 * Replace the 3 by 3 area by diamonds.
124 	 */
125 	public void blow() {
126 		this.replaceByDiamond();
127 		this.lookAtNextBlock(Direction.Up).replaceByDiamond();
128 		this.lookAtNextBlock(Direction.Left).replaceByDiamond();
129 		this.lookAtNextBlock(Direction.Down).replaceByDiamond();
130 		this.lookAtNextBlock(Direction.Right).replaceByDiamond();
131 		this.lookAtNextBlock(Direction.RightUp).replaceByDiamond();
132 		this.lookAtNextBlock(Direction.RightDown).replaceByDiamond();
133 		this.lookAtNextBlock(Direction.LeftDown).replaceByDiamond();
134 		this.lookAtNextBlock(Direction.LeftUp).replaceByDiamond();
135 	}
136 
137 }
